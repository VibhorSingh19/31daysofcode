#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int T;
   cin>>T;
   while(T--)
   {
       int N,s=0;
       cin>>N;
       int arr[N];
       for(int i=0;i<N;i++)
       {
           cin>>arr[i];
       }
       for(int i=0;i<N;i++)
       {
           s+=arr[i];
       }
       int s1=0,s2=0,x=0;
       s1+=arr[x++];
       while(s1+s2==s)
       {
        if(s2>s1)
        s1+=arr[x++];
        if(s1>s2)
        s2+=arr[x++];
       }
       if(s2>s1)
       cout<<s2;
       else
       cout<<s1;
       
       
   }

    return 0;
}
