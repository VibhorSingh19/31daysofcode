
#include <iostream>
#include <bits/stdc++.h> 
using namespace std;

int main()
{
    int N;
    cin>>N;
    int arr[N];
    for(int i=0;i<N;i++)
    {
     cin>>arr[i];   
    }
    int n = sizeof(arr)/sizeof(arr[0]); 
    sort(arr, arr+n);
    int x,y=1,max=0;
    for(int i=0;i<N;i++)
    {
        x=N-i;
        y=arr[i]*x;
        if(y>max)
        max=y;
    }
    cout<<max;
    cout<<endl;
       
    return 0;
}
